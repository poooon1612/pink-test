<!--

🎨 WebAR 学習テンプレート Ver.1.0


このテンプレートは

「作品を画像認識して説明を表示する」

ための学習用テンプレート。


作品ごとに主に変更する場所

・title

・targets.mind

・value（練習時）

・画像・動画ファイル名

・position

・rotation

・scale

-->




<!DOCTYPE html>

<html lang="ja">


<!-- ======================================== -->

<!-- 🏠 第2章　head（裏方スタッフ） -->

<!-- 展示を始める前の準備をする場所 -->

<!-- 基本は変更しません -->

<!-- ======================================== -->


<head>


  <!-- 🌏 世界中の文字（日本語・英語・絵文字）を使えるようにする -->

  <meta charset="UTF-8">


  <!-- 📱 iPhoneやAndroidの画面サイズに合わせる -->

  <meta name="viewport" content="width=device-width, initial-scale=1.0">


  <!-- 🏷️ ブラウザのタブに表示される作品名 -->

  <title>Pink AR Test</title>


  <!-- ======================================== -->

  <!-- 🔌 第3章　ARのエンジンを読み込む -->

  <!-- 基本は変更しません -->

  <!-- ======================================== -->


  <!-- 🎨 A-Frame（AR空間を作る） -->

  <script src="https://aframe.io/releases/1.4.2/aframe.min.js"></script>


  <!-- 📷 MindAR（作品画像を認識する） -->

  <script src="https://cdn.jsdelivr.net/npm/mind-ar@1.2.5/dist/mindar-image-aframe.prod.js"></script>


</head>


<body>


 <!-- ======================================== -->

 <!-- 🏛 第4章　AR展示室 -->

 <!-- ======================================== -->


  <!-- ======================================== -->

  <!-- 🚪 第5章　来場者の入口（カメラ） -->

  <!-- スマホのカメラを使って作品を探します -->

  <!-- 基本は変更しません -->

  <!-- ======================================== -->


  <a-scene

    mindar-image="imageTargetSrc: targets.mind;"

    color-space="sRGB"

    renderer="colorManagement: true;"

    vr-mode-ui="enabled: false"

    device-orientation-permission-ui="enabled: true">


  <!-- ======================================== -->

  <!-- 🖼 第6章　作品を認識したら置くもの -->

  <!-- Blenderでいうオブジェクト追加 -->

  <!-- ======================================== -->


    <a-entity mindar-image-target="targetIndex: 0">


    <!-- ======================================== -->

    <!-- 💬 第7章　最初の表示テスト -->

    <!-- ======================================== -->


      <a-text

        value="HELLO!"

        color="white"

        align="center"

        position="0 0 0"

        width="2"

      ></a-text>


    </a-entity>


  </a-scene>


</body>


</html>

src        → 読み込む画像・動画

color      → 色

height     → 高さ
